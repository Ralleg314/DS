@startuml

legend top 
    DS2 Visualitzar Catàleg
end legend

actor Usuari as usuari
activate usuari

participant ":TotSeries" as TS
participant ":Dades" as dades
participant ":Serie" as serie
participant ":Temporada" as temporada
participant ":Episodi" as episodi

usuari ->TS: veureCataleg()
activate TS

TS -> dades: visualitzarCataleg(): Catàleg
activate dades
loop
    |||
    dades ->serie : visualitzarSerie()
    activate serie
    loop
        |||
        serie ->temporada : visualitzarTemporada()
        activate temporada
        loop
            |||
            temporada -> episodi : visualitzarEpisodi()
            activate episodi
            episodi --> temporada : Episodi
            deactivate episodi
            |||
        end
        temporada --> serie : Temporada
        deactivate temporada
        |||
    end
    serie --> dades : Serie
    deactivate serie
    |||
end
dades --> TS : Cataleg
deactivate dades
TS --> usuari
deactivate TS
deactivate usuari


@enduml
