@startuml

legend top 
    DS2 Visualitzar Catàleg
end legend

actor Usuari as usuari
activate usuari

participant ":Menu" as menu
participant ":TotSeries" as TS
participant ":Dades" as dades
participant ":Serie" as serie

usuari -> menu
activate menu
menu ->menu : op=llegeixInt()
opt op==2
    |||
    menu ->TS: visualitzarCataleg()
    activate TS
    
    TS -> dades: visualitzarCataleg(): Catàleg
    activate dades
    loop series
        |||
        dades ->serie : toString()
        activate serie
        serie --> dades : String
        deactivate serie
        |||
    end
    dades --> TS : cataleg=list<String>
    deactivate dades
    TS ---> menu : cataleg
    deactivate TS
    loop cataleg
        |||
        menu->menu : escriu(cataleg)
        |||
    end
    deactivate menu
    menu --> usuari
    deactivate TS
    deactivate usuari
    |||
end

@enduml
